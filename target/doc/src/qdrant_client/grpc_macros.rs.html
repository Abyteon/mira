1/// Default type conversions between builders and their built types. This allows us not writing .build() all the time, especially if we use builders within buliders.
2///
3/// This also adds a build() function for all builders. The build() function automatically generated by derive_builder returns
4/// a Result which is ugly to have to unwrap() all the time, especially because they can't fail in our case.
5macro_rules! builder_type_conversions {
6    ($main_type:ident,$builder_type:ident,true) => {
7        builder_type_conversions!($main_type, $builder_type, build_inner);
8
9        // An empty() function that allows crate-wide instantiation of empty builder because
10        // the create_empty() function by `derive_builder` is non-configurable and private only.
11        impl $builder_type {
12            pub(crate) fn empty() -> Self {
13                Self::create_empty()
14            }
15        }
16    };
17
18    ($main_type:ident,$builder_type:ident,$build_fn:ident) => {
19        impl From<$builder_type> for $main_type {
20            fn from(value: $builder_type) -> Self {
21                value.$build_fn().expect(&format!(
22                    "Failed to convert {} to {}",
23                    stringify!($builder_type),
24                    stringify!($main_type)
25                ))
26            }
27        }
28
29        impl $builder_type {
30            /// Builds the desired type. Can often be omitted.
31            pub fn build(self) -> $main_type {
32                self.$build_fn().expect(&format!(
33                    "Failed to build {} into {}",
34                    stringify!($builder_type),
35                    stringify!($main_type)
36                ))
37            }
38        }
39    };
40
41    // Small shortcut so we don't have to write build_inner for most of the types.
42    ($main_type:ident,$builder_type:ident) => {
43        builder_type_conversions!($main_type, $builder_type, build_inner);
44    };
45}
46pub(crate) use builder_type_conversions;

48/// Helper function to convert from &Option<T> to Option<U> if U: From<T> is satisfied.
49pub(crate) fn convert_option<T, U>(input: &Option<T>) -> Option<U>
50where
51    U: From<T>,
52    T: Clone,
53{
54    let input = input.as_ref()?;
55    Some(input.clone().into())
56}

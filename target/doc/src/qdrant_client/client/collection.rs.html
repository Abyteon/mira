26    pub async fn with_collections_client<T, O: Future<Output = anyhow::Result<T, Status>>>(
27        &self,
28        f: impl Fn(CollectionsClient<InterceptedService<Channel, TokenInterceptor>>) -> O,
29    ) -> anyhow::Result<T, Status> {
30        self.channel
31            .with_channel(
32                |channel| {
33                    let service = self.with_api_key(channel);
34                    let mut client =
35                        CollectionsClient::new(service).max_decoding_message_size(usize::MAX);
36                    if let Some(compression) = self.cfg.compression {
37                        client = client
38                            .send_compressed(compression.into())
39                            .accept_compressed(compression.into());
40                    }
41                    f(client)
42                },
43                false,
44            )
45            .await
46    }

65    pub async fn has_collection(&self, collection_name: impl ToString) -> anyhow::Result<bool> {
66        let collection_name = collection_name.to_string();
67        let response = self.list_collections().await?;
68        let result = response
69            .collections
70            .into_iter()
71            .any(|c| c.name == collection_name);
72
73        Ok(result)
74    }

80    pub async fn collection_exists(&self, collection_name: impl ToString) -> anyhow::Result<bool> {
81        let collection_name_ref = &collection_name.to_string();
82        Ok(self
83            .with_collections_client(|mut collection_api| async move {
84                let request = CollectionExistsRequest {
85                    collection_name: collection_name_ref.clone(),
86                };
87                let result = collection_api.collection_exists(request).await?;
88                Ok(result
89                    .into_inner()
90                    .result
91                    .map(|r| r.exists)
92                    .unwrap_or(false))
93            })
94            .await?)
95    }

213            actions: vec![AliasOperations {

216            timeout: None,
217        };

219    }
